// QUICK START

Up and Running
in 3 Minutes

# 1. Clone the repo
git clone https://github.com/yourusername/smartx-backend.git
cd smartx-backend

# 2. Install dependencies
npm install

# 3. Set up environment variables
cp .env.example .env
# Edit .env with your MongoDB, Groq, Cloudinary keys

# 4. Start development server
npm run dev

# ✅ Server running at http://localhost:5000
# ✅ Socket.io ready on same port
// POST /api/auth/register
const response = await fetch('/api/auth/register', {
  method: 'POST',
  headers: { 'Content-Type': 'application/json' },
  body: JSON.stringify({
    name: 'John Doe',
    email: 'john@example.com',
    password: 'securepassword123'
  })
});

// Response:
// { success: true, message: "Verify your email", userId: "..." }

// POST /api/auth/verify-otp
const verify = await fetch('/api/auth/verify-otp', {
  method: 'POST',
  body: JSON.stringify({ userId, otp: '123456' })
});
// Returns: accessToken + refreshToken
import { io } from 'socket.io-client';

const socket = io('http://localhost:5000', {
  auth: { token: 'your_jwt_token' }
});

// Send a message
socket.emit('message:send', {
  chatId: 'chat_id_here',
  content: 'Hello! Is this product available?'
});

// Receive messages
socket.on('message:received', ({ chatId, message }) => {
  console.log(`New msg from ${message.sender.name}`);
});

// Typing indicators
socket.emit('typing:start', { chatId });
socket.emit('typing:stop', { chatId });
// POST /api/ai/chat — Chat with Llama 3.3
const res = await fetch('/api/ai/chat', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': `Bearer ${accessToken}`
  },
  body: JSON.stringify({
    message: 'Find me a good laptop under ₹50,000',
    context: [] // optional conversation history
  })
});

// Response:
// { success: true, response: "Here are the best options..." }

// POST /api/ai/tags — Generate product tags
const tags = await fetch('/api/ai/tags', {
  method: 'POST',
  body: JSON.stringify({
    title: 'Gaming Keyboard',
    category: 'electronics'
  })
});
// Returns: ["mechanical", "rgb", "gaming", ...]
